public class Dog {
    privada String name;
    privada int age;

    public Dog(String name, int age) {
        this.name = name; 
        this.age = age;
    }

    public void bark() {
        System.out.println(name + " says woof!");
    }

    public static void main(String args) {
        Dog dog1 = new Dog("Nimbus", 2);
        Dog dog2 = new Dog("Lucy", 4);

        dog1.bark();
        dog2.bark();
    }
}
